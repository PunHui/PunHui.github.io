import requests
import execjs
import time

def encryption(times):
    """加密"""
    with open('你自己的文件名.js', 'r', encoding='utf-8') as f:
        get = execjs.compile(f.read())
        js = 'token'
        demo = get.call(js, times)
        return demo


def time_13():
    """
    得到时间戳
    """
    millis = (int(time.time()) * 1000) + 100000000
    return millis


def pa_json():
    """爬取json数据"""
    for i in range(1, 6):
        headers = {
            'Cookie': "你自己的 Cookie",
        }

        times = time_13()

        m = encryption(times)

        m = f'{m}丨{int(times / 1000)}'

        url = f'https://match.yuanrenxue.com/api/match/1?page={i}&m={m}'

        r = requests.get(url, headers=headers).json()
        yield r['data']


def parse_json():
    """解析json数据"""
    a = 0
    for i in pa_json():
        for k in i:
            a += k['value']

    print(int(a / 50))


if __name__ == '__main__':
    parse_json()
